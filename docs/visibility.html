	$addClass(el, 'animate-slide-in-right');
	$addClass(el, 'animate-2s');
	$addClass(el, 'animate-when-visible');
});

$forEach(paragraphs, function (el) {
	$addClass(el, 'animate-fade-in');
	$addClass(el, 'animate-2s');
	$addClass(el, 'animate-when-visible');
});

$visibility('.animate-when-visible', {invisibleClass: 'reset-animation'}, function (el, isVisibile) {
	console.log(isVisibile ? '[visible]' : '[invisible]', el.tagName);
});

The visibility tool will use the scroll event to track element position inside the viewport. It automatically adds/remove classes from the element or you can add your own callback to manipulate it.
